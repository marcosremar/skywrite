% Command to create a colored block with a title
\newcommand{\commentblock}[2]{%
  \ifshowcommentblocks%
    \stepcounter{colorcount}%
    \ifnum\value{colorcount}>5\setcounter{colorcount}{1}\fi%
    \def\currentcolor{pastel\arabic{colorcount}}%
    \begin{tcolorbox}[
      colback=\currentcolor,
      colframe=\currentcolor!80!black,
      title=#1,
      fonttitle=\bfseries
    ]
    #2
    \end{tcolorbox}
  \fi%
}

% Command to toggle comment blocks on/off

}
